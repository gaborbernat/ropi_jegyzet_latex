% Smiley commands
\newcommand{\smiley}{\tikz[baseline=-0.75ex,ColorGreyish]{
    \draw circle (2mm);
\node[fill,circle,inner sep=0.5pt] (left eye) at (135:0.8mm) {};
\node[fill,circle,inner sep=0.5pt] (right eye) at (45:0.8mm) {}; \draw
(-145:0.9mm) arc (-120:-60:1.5mm);
    }
}
